title: Setup Development Environment
sidebar_position: 2
---

You must enable the dev tools module in the client SDK to test and develop plugins locally. This adds `Localhost Dev` to the plugins tab's list of plugins. 

You can run your plugin application on `localhost:5000` and view it by opening the `Localhost Dev` plugin in your meeting.

[//]: # (TODO:ishita1805 - Tabs for all distributions of web-core)

:::info before you start
Install `@dytesdk/react-web-core@0.10.x` or higher to use dev tools.
:::

```ts
import { useDyteClient } from '@dytesdk/react-web-core';
import { DyteMeeting } from '@dytesdk/react-ui-kit';

export default function App() {
  const [meeting, initMeeting] = useDyteClient();

  useEffect(() => {
    initMeeting({
      roomName: '<room-name>',
      authToken: '<auth-token>',
      defaults: {
        audio: false,
        video: false,
      },
      modules: {
        devTools: true,
      }
    });
  }, []);

  return (
      <DyteMeeting mode="fill" meeting={meeting} />
  );
}

```
