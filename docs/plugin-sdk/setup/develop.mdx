In order to develop your plugin, you need to setup your development environment. You can open your plugin inside a dyte meeting in order to develop it. 

For this you would need to have a dyte meeting setup first. Dyte meetings use the client SDKs. 

The code snippet below shows how you can quickly setup a dyte meeting.

You must enable the dev tools module in the client SDK. This adds `Localhost Dev` to the list of plugins. You need to run your plugin application on `localhost:5000` and open the `Localhost Dev` plugin to view your plugin.

:::info before you start
  Install `@dytesdk/web-core@0.10.x` or higher to use dev tools.
:::

```ts
<body>
  <dyte-meeting id="my-meeting"></dyte-meeting>

  <script>
    const init = async () => {
      const meeting = await DyteClient.init({
        authToken: '',
        roomName: '',
        defaults: {
          audio: true,
          video: true,
        },
      });

      document.getElementById('my-meeting').meeting = meeting;
    };

    init();
  </script>
</body>
```

<TabItem value="react">

:::info before you start
  Install `@dytesdk/react-web-core@0.10.x` or higher to use dev tools.
:::

```ts
import { useDyteClient } from '@dytesdk/react-web-core';
import { DyteMeeting } from '@dytesdk/react-ui-kit';

export default function App() {
  const [meeting, initMeeting] = useDyteClient();

  useEffect(() => {
    initMeeting({
      roomName: '<room-name>',
      authToken: '<auth-token>',
      modules: {
        devTools: true,
      }
    });
  }, []);

  return (
      <DyteMeeting mode="fill" meeting={meeting} />
  );
}
```

<TabItem value="angular">

:::info before you start
  Install `@dytesdk/angular-web-core@0.10.x` or higher to use dev tools.
:::

Put the dyte component in component.html file

```ts
<dyte-meeting #myid></dyte-meeting>
```

Then in the component.ts or component.js file initialize the meeting

```ts
class AppComponent {
  title = 'MyProject';
  @ViewChild('myid') meetingComponent: DyteMeeting;
  dyteMeeting: DyteClient;

  async ngAfterViewInit() {
    const meeting = await DyteClient.init({
      roomName: '<room-name>',
      authToken: '<auth-token>',
      modules: {
        devTools: true,
      }
    });
    meeting.joinRoom();
    this.dyteMeeting = meeting;
    if (this.meetingComponent) this.meetingComponent.meeting = meeting;
  }
}
```
